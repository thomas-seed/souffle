101 /**
102  * The generic base type of all RamVisitors realizing the dispatching of
103  * visitor calls. Each visitor may define a return type R and a list of
104  * extra parameters to be passed along with the visited RamNodes to the
105  * corresponding visitor function.
106  *
107  * @tparam R the result type produced by a visit call
108  * @tparam Params extra parameters to be passed to the visit call
109  */

125  /**
126  * The main entry for a visit process conducting the dispatching of
127  * a visit to the various sub-types of RamNodes. Sub-classes may override
128  * this implementation to conduct pre-visit operations.
129  *
130  * Note that the order of this list is important. Sub-classes must be listed

320 /**
321  * A utility function visiting all nodes within the RAM fragment rooted by the given node
322  * recursively in a depth-first pre-order fashion applying the given visitor to each
323  * encountered node.
324  *
325  * @param root the root of the RAM fragment to be visited
326  * @param visitor the visitor to be applied on each node
327  * @param args a list of extra parameters to be forwarded to the visitor
328  */
329 template <typename R, typename... Ps, typename... Args>
330 void visitDepthFirstPreOrder(const RamNode& root, RamVisitor<R, Ps...>& visitor, Args&... args) {
331  visitor(root, args...);
332  for (const RamNode* cur : root.getChildNodes()) {
333  if (cur != nullptr) {
334  visitDepthFirstPreOrder(*cur, visitor, args...);
335  }
336  }
337 }

355 /**
356  * A specialized visitor wrapping a lambda function -- an auxiliary type required
357  * for visitor convenience functions.
358  */
359 template <typename R, typename N>
360 struct LambdaRamVisitor : public RamVisitor<void> {
361  std::function<R(const N&)> lambda;
362  LambdaRamVisitor(std::function<R(const N&)> lambda) : lambda(std::move(lambda)) {}
363  void visit(const RamNode& node) override {
364  if (const auto* n = dynamic_cast<const N*>(&node)) {
365  lambda(*n);
366  }
367  }
368 };
369 
370 /**
371  * A factory function for creating LambdaRamVisitor instances.
372  */
373 template <typename R, typename N>
374 LambdaRamVisitor<R, N> makeLambdaRamVisitor(const std::function<R(const N&)>& fun) {
375  return LambdaRamVisitor<R, N>(fun);
376 }
