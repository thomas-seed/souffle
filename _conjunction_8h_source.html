33 /**
34  * @class RamConjunction
35  * @brief A conjunction of conditions
36  *
37  * Condition of the form "LHS and RHS", where LHS
38  * and RHS are conditions
39  *
40  * For example:
41  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
42  * C1 AND C2 AND C3
43  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
44  * Is a RamConjunction, which may have LHS "C1"
45  * and RHS "C2 AND C3"
46  */
47 class RamConjunction : public RamCondition {
48 public:
49  RamConjunction(std::unique_ptr<RamCondition> l, std::unique_ptr<RamCondition> r)
50  : lhs(std::move(l)), rhs(std::move(r)) {
51  assert(lhs != nullptr && "left-hand side of conjunction is a nullptr");
52  assert(rhs != nullptr && "right-hand side of conjunction is a nullptr");
53  }
54 
55  /** @brief Get left-hand side of conjunction */
56  const RamCondition& getLHS() const {
57  return *lhs;
58  }
59 
60  /** @brief Get right-hand side of conjunction */
61  const RamCondition& getRHS() const {
62  return *rhs;
63  }
64 
65  std::vector<const RamNode*> getChildNodes() const override {
66  return {lhs.get(), rhs.get()};
67  }
68 
69  RamConjunction* clone() const override {
70  return new RamConjunction(souffle::clone(lhs), souffle::clone(rhs));
71  }
72 
73  void apply(const RamNodeMapper& map) override {
74  lhs = map(std::move(lhs));
75  rhs = map(std::move(rhs));
