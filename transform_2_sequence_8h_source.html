29 /**
30  * @Class RamTransformerSequence
31  * @Brief Composite sequence transformer
32  *
33  * A sequence of transformations is applied to a translation unit
34  * sequentially. The last transformation decides the outcome whether
35  * the code has been changed.
36  *
37  */
38 class RamTransformerSequence : public RamMetaTransformer {
39 public:
40  template <typename... Tfs>
41  RamTransformerSequence(std::unique_ptr<Tfs>&&... tf) : RamTransformerSequence() {
42  std::unique_ptr<RamTransformer> tmp[] = {std::move(tf)...};
43  for (auto& cur : tmp) {
44  transformers.emplace_back(std::move(cur));
45  }
46  for (const auto& cur : transformers) {
47  (void)cur;
48  assert(cur);
49  }
50  }
51  RamTransformerSequence() = default;
52  std::string getName() const override {
53  return "RamTransformerSequence";
54  }
55  bool transform(RamTranslationUnit& tU) override {
56  bool changed = false;
57  // The last transformer decides the status
58  // of the change flag.
59  // Note that for other semantics, new transformer
60  // sequence class needs to be introduced.
61  for (auto const& cur : transformers) {
62  changed = cur->apply(tU);
63  }
64  return changed;
65  }
66 
67 protected:
68  /** sequence of transformers */
69  std::vector<std::unique_ptr<RamTransformer>> transformers;
70 };
