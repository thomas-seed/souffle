35 /**
36  * @class RamProject
37  * @brief Project a result into the target relation.
38  *
39  * For example:
40  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
41  * FOR t0 IN A
42  * ...
43  * PROJECT (t0.a, t0.b, t0.c) INTO @new_X
44  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
45  */
46 class RamProject : public RamOperation {
47 public:
48  RamProject(std::unique_ptr<RamRelationReference> relRef,
49  std::vector<std::unique_ptr<RamExpression>> expressions)
50  : relationRef(std::move(relRef)), expressions(std::move(expressions)) {
51  assert(relationRef != nullptr && "Relation reference is a null-pointer");
52  for (auto const& expr : expressions) {
53  assert(expr != nullptr && "Expression is a null-pointer");
54  }
55  }
56 
57  /** @brief Get relation */
58  const RamRelation& getRelation() const {
59  return *relationRef->get();
60  }
61 
62  /** @brief Get expressions */
63  std::vector<RamExpression*> getValues() const {
64  return toPtrVector(expressions);
65  }
66 
67  std::vector<const RamNode*> getChildNodes() const override {
68  std::vector<const RamNode*> res;
69  res.push_back(relationRef.get());
70  for (const auto& expr : expressions) {
71  res.push_back(expr.get());
72  }
73  return res;
74  }
75 
76  RamProject* clone() const override {
77  std::vector<std::unique_ptr<RamExpression>> newValues;
78  for (auto& expr : expressions) {
79  newValues.emplace_back(expr->clone());
80  }
81  return new RamProject(souffle::clone(relationRef), std::move(newValues));
82  }
83 
84  void apply(const RamNodeMapper& map) override {
85  relationRef = map(std::move(relationRef));
86  for (auto& expr : expressions) {
87  expr = map(std::move(expr));
88  }
89  }
90 
91 protected:
92  void print(std::ostream& os, int tabpos) const override {
93  os << times(" ", tabpos);
94  os << "PROJECT (" << join(expressions, ", ", print_deref<std::unique_ptr<RamExpression>>())
95  << ") INTO " << getRelation().getName() << std::endl;
96  }
97 
98  bool equal(const RamNode& node) const override {
99  const auto& other = static_cast<const RamProject&>(node);
100  return equal_ptr(relationRef, other.relationRef) && equal_targets(expressions, other.expressions);
101  }
102 
103  /** Relation that values are projected into */
104  std::unique_ptr<RamRelationReference> relationRef;
105 
106  /* Values (expressions) for projection */
107  std::vector<std::unique_ptr<RamExpression>> expressions;
108 };
