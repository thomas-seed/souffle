36 NormalisedClause::NormalisedClause(const AstClause* clause) {
37  // head
38  AstQualifiedName name("@min:head");
39  std::vector<std::string> headVars;
40  for (const auto* arg : clause->getHead()->getArguments()) {
41  headVars.push_back(normaliseArgument(arg));
42  }
43  clauseElements.push_back({.name = name, .params = headVars});
44 
45  // body
46  for (const auto* lit : clause->getBodyLiterals()) {
47  addClauseBodyLiteral("@min:scope:0", lit);
48  }
49 }

64 void NormalisedClause::addClauseBodyLiteral(const std::string& scopeID, const AstLiteral* lit) {
65  if (const auto* atom = dynamic_cast<const AstAtom*>(lit)) {
66  addClauseAtom("@min:atom", scopeID, atom);
67  } else if (const auto* neg = dynamic_cast<const AstNegation*>(lit)) {
68  addClauseAtom("@min:neg", scopeID, neg->getAtom());
69  } else if (const auto* bc = dynamic_cast<const AstBinaryConstraint*>(lit)) {
70  AstQualifiedName name(toBinaryConstraintSymbol(bc->getOperator()));
71  name.prepend("@min:operator");
72  std::vector<std::string> vars;
73  vars.push_back(scopeID);
74  vars.push_back(normaliseArgument(bc->getLHS()));
75  vars.push_back(normaliseArgument(bc->getRHS()));
76  clauseElements.push_back({.name = name, .params = vars});
77  } else {
78  assert(lit != nullptr && "unexpected nullptr lit");
79  fullyNormalised = false;
80  std::stringstream qualifier;
81  qualifier << "@min:unhandled:lit:" << scopeID;
82  AstQualifiedName name(toString(*lit));
83  name.prepend(qualifier.str());
84  clauseElements.push_back({.name = name, .params = std::vector<std::string>()});
85  }
86 }

88 std::string NormalisedClause::normaliseArgument(const AstArgument* arg) {
89  if (auto* stringCst = dynamic_cast<const AstStringConstant*>(arg)) {
90  std::stringstream name;
91  name << "@min:cst:str" << *stringCst;
92  constants.insert(name.str());
93  return name.str();
94  } else if (auto* numericCst = dynamic_cast<const AstNumericConstant*>(arg)) {
95  std::stringstream name;
96  name << "@min:cst:num:" << *numericCst;
97  constants.insert(name.str());
98  return name.str();
99  } else if (dynamic_cast<const AstNilConstant*>(arg) != nullptr) {
100  constants.insert("@min:cst:nil");
101  return "@min:cst:nil";
102  } else if (auto* var = dynamic_cast<const AstVariable*>(arg)) {
103  auto name = var->getName();
104  variables.insert(name);
105  return name;
106  } else if (dynamic_cast<const AstUnnamedVariable*>(arg)) {
107  static size_t countUnnamed = 0;
108  std::stringstream name;
109  name << "@min:unnamed:" << countUnnamed++;
110  variables.insert(name.str());
111  return name.str();
112  } else if (auto* aggr = dynamic_cast<const AstAggregator*>(arg)) {
113  // Set the scope to uniquely identify the aggregator
114  std::stringstream scopeID;
115  scopeID << "@min:scope:" << ++aggrScopeCount;
116  variables.insert(scopeID.str());
117 
118  // Set the type signature of this aggregator
119  std::stringstream aggrTypeSignature;
120  aggrTypeSignature << "@min:aggrtype";
121  std::vector<std::string> aggrTypeSignatureComponents;
122 
123  // - the operator is fixed and cannot be changed
124  aggrTypeSignature << ":" << aggr->getOperator();
125 
126  // - the scope can be remapped as a variable
127  aggrTypeSignatureComponents.push_back(scopeID.str());
128 
129  // - the normalised target expression can be remapped as a variable
130  if (aggr->getTargetExpression() != nullptr) {
131  std::string normalisedExpr = normaliseArgument(aggr->getTargetExpression());
132  aggrTypeSignatureComponents.push_back(normalisedExpr);
133  }
134 
135  // Type signature is its own special atom
136  clauseElements.push_back({.name = aggrTypeSignature.str(), .params = aggrTypeSignatureComponents});
137 
138  // Add each contained normalised clause literal, tying it with the new scope ID
139  for (const auto* literal : aggr->getBodyLiterals()) {
140  addClauseBodyLiteral(scopeID.str(), literal);
141  }
142 
143  // Aggregator identified by the scope ID
144  return scopeID.str();
145  } else {
146  fullyNormalised = false;
147  return "@min:unhandled:arg";
148  }
149 }

151 void ClauseNormalisationAnalysis::run(const AstTranslationUnit& translationUnit) {
152  const auto& program = *translationUnit.getProgram();
153  for (const auto* clause : program.getClauses()) {
154  assert(!contains(normalisations, clause) && "clause already processed");
155  normalisations[clause] = NormalisedClause(clause);
156  }
157 }
