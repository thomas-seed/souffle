45 public:
46  RamIndexOperation(std::unique_ptr<RamRelationReference> r, int ident, RamPattern queryPattern,
47  std::unique_ptr<RamOperation> nested, std::string profileText = "")
48  : RamRelationOperation(std::move(r), ident, std::move(nested), std::move(profileText)),
49  queryPattern(std::move(queryPattern)) {
50  assert(getRangePattern().first.size() == getRelation().getArity() &&
51  getRangePattern().second.size() == getRelation().getArity());
52  for (const auto& pattern : queryPattern.first) {
53  assert(pattern != nullptr && "pattern is a null-pointer");
54  }
55  for (const auto& pattern : queryPattern.second) {
56  assert(pattern != nullptr && "pattern is a null-pointer");
57  }

60  /**
61  * @brief Get range pattern
62  * @return A pair of std::vector of pointers to RamExpression objects
63  * These vectors represent the lower and upper bounds for each attribute in the tuple
64  * <expr1> <= Tuple[level, element] <= <expr2>
65  * We have that at an index for an attribute, its bounds are <expr1> and <expr2> respectively
66  * */
67  std::pair<std::vector<RamExpression*>, std::vector<RamExpression*>> getRangePattern() const {
68  return std::make_pair(toPtrVector(queryPattern.first), toPtrVector(queryPattern.second));
69  }

93  RamPattern resQueryPattern;

104  /** @brief Helper method for printing */
105  void printIndex(std::ostream& os) const {
106  // const auto& attrib = getRelation().getAttributeNames();
107  bool first = true;
108  for (unsigned int i = 0; i < getRelation().getArity(); ++i) {
109  // TODO: print proper upper lower/bound
110 
111  // early exit if no upper/lower bounds are defined
112  if (isRamUndefValue(queryPattern.first[i].get()) &&
113  isRamUndefValue(queryPattern.second[i].get())) {
114  continue;
115  }
116 
117  if (first) {
118  os << " ON INDEX ";
119  first = false;
120  } else {
121  os << " AND ";
122  }
123 
124  // both bounds defined and equal => equality
125  if (!isRamUndefValue(queryPattern.first[i].get()) &&
126  !isRamUndefValue(queryPattern.second[i].get())) {
127  // print equality when lower bound = upper bound
128  if (*(queryPattern.first[i]) == *(queryPattern.second[i])) {
129  os << "t" << getTupleId() << ".";
130  os << i << " = ";
131  os << *(queryPattern.first[i]);
132  continue;
133  }
134  }
135  // at least one bound defined => inequality
136  if (!isRamUndefValue(queryPattern.first[i].get()) ||
137  !isRamUndefValue(queryPattern.second[i].get())) {
138  if (!isRamUndefValue(queryPattern.first[i].get())) {
139  os << *(queryPattern.first[i]) << " <= ";
140  }
141 
142  os << "t" << getTupleId() << ".";
143  os << i;
144 
145  if (!isRamUndefValue(queryPattern.second[i].get())) {
146  os << " <= " << *(queryPattern.second[i]);
147  }
148 
149  continue;
150  }
151  }
152  }
