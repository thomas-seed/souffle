35 /**
36  * @class RamConstraint
37  * @brief Evaluates a binary constraint with respect to two RamExpressions
38  *
39  * Condition is true if the constraint (a logical operator
40  * such as "<") holds between the two operands
41  *
42  * The following example checks the equality of
43  * the two given tuple elements:
44  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
45  * t0.1 = t1.0
46  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
47  */
48 class RamConstraint : public RamCondition {
49 public:
50  RamConstraint(BinaryConstraintOp op, std::unique_ptr<RamExpression> l, std::unique_ptr<RamExpression> r)
51  : op(op), lhs(std::move(l)), rhs(std::move(r)) {
52  assert(lhs != nullptr && "left-hand side of constraint is a null-pointer");
53  assert(rhs != nullptr && "right-hand side of constraint is a null-pointer");
54  }
55 
56  /** @brief Get left-hand side */
57  const RamExpression& getLHS() const {
58  return *lhs;
59  }
60 
61  /** @brief Get right-hand side */
62  const RamExpression& getRHS() const {
63  return *rhs;
64  }
65 
66  /** @brief Get operator symbol */
67  BinaryConstraintOp getOperator() const {
68  return op;
69  }
70 
71  std::vector<const RamNode*> getChildNodes() const override {
72  return {lhs.get(), rhs.get()};
73  }
74 
75  RamConstraint* clone() const override {
76  return new RamConstraint(op, souffle::clone(lhs), souffle::clone(rhs));
77  }
78 
79  void apply(const RamNodeMapper& map) override {
80  lhs = map(std::move(lhs));
81  rhs = map(std::move(rhs));
82  }
83 
84 protected:
85  void print(std::ostream& os) const override {
86  os << "(" << *lhs << " ";
87  os << toBinaryConstraintSymbol(op);
88  os << " " << *rhs << ")";
89  }
90 
91  bool equal(const RamNode& node) const override {
92  const auto& other = static_cast<const RamConstraint&>(node);
93  return op == other.op && equal_ptr(lhs, other.lhs) && equal_ptr(rhs, other.rhs);
94  }
