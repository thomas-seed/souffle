36 /**
37  * @class RamUnpackRecord
38  * @brief Record lookup
39  *
40  * Looks up a record with respect to an expression
41  *
42  * For example:
43  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
44  * UNPACK t0.0 INTO t1
45  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
46  */
47 class RamUnpackRecord : public RamTupleOperation {
48 public:
49  RamUnpackRecord(std::unique_ptr<RamOperation> nested, int ident, std::unique_ptr<RamExpression> expr,
50  size_t arity)
51  : RamTupleOperation(ident, std::move(nested)), expression(std::move(expr)), arity(arity) {
52  assert(expression != nullptr && "Expression is a null-pointer");
53  }
54 
55  /** @brief Get record expression */
56  const RamExpression& getExpression() const {
57  assert(expression != nullptr && "Expression of unpack-record is a null-pointer");
58  return *expression;
59  }
60 
61  /** @brief Get arity of record */
62  std::size_t getArity() const {
63  return arity;
64  }
65 
66  std::vector<const RamNode*> getChildNodes() const override {
67  auto res = RamTupleOperation::getChildNodes();
68  res.push_back(expression.get());
69  return res;
70  }
71 
72  RamUnpackRecord* clone() const override {
73  return new RamUnpackRecord(
74  souffle::clone(&getOperation()), getTupleId(), souffle::clone(&getExpression()), arity);
75  }
76 
77  void apply(const RamNodeMapper& map) override {
78  RamTupleOperation::apply(map);
79  expression = map(std::move(expression));
80  }
81 
82 protected:

84  os << times(" ", tabpos);
85  os << "UNPACK t" << getTupleId() << " FROM " << *expression << "\n";
86  RamNestedOperation::print(os, tabpos + 1);
87  }
88 
89  bool equal(const RamNode& node) const override {
90  const auto& other = static_cast<const RamUnpackRecord&>(node);
91  return RamTupleOperation::equal(other) && equal_ptr(expression, other.expression) &&

93  }
