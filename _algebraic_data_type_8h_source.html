26 /**
27  * @class AstAlgebraicDataType
28  * @brief Combination of types using sums and products.
29  *
30  * ADT combines a simpler types using product types and sum types.
31  *
32  * Example:
33  * .type Nat = S {n : Nat} | Zero {}
34  *
35  * The type Nat has two branches, S which takes element of type Nat and Zero which doesn't take any
36  * arguments.
37  *
38  */
39 class AstAlgebraicDataType : public AstType {
40 public:
41  AstAlgebraicDataType(AstQualifiedName name, VecOwn<AstBranchDeclaration> branches, SrcLocation loc = {})
42  : AstType(std::move(name), std::move(loc)), branches(std::move(branches)) {
43  assert(!this->branches.empty());
44  };
45 
46  std::vector<AstBranchDeclaration*> getBranches() const {
47  return toPtrVector(branches);
48  }
49 
50  void print(std::ostream& os) const override {
51  os << tfm::format(".type %s = %s", getQualifiedName(), join(branches, " | "));
52  }
53 
54  AstAlgebraicDataType* clone() const override {
55  return new AstAlgebraicDataType(getQualifiedName(), souffle::clone(branches), getSrcLoc());
56  }
57 
58 protected:
59  bool equal(const AstNode& node) const override {
60  const auto& other = dynamic_cast<const AstAlgebraicDataType&>(node);
61  return getQualifiedName() == other.getQualifiedName() && branches == other.branches;
62  }
63 
64 private:
65  /** The list of branches for this sum type. */
66  VecOwn<AstBranchDeclaration> branches;
67 };
