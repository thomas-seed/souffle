40 class RamAbstractExistenceCheck : public RamCondition {

42  RamAbstractExistenceCheck(
43  std::unique_ptr<RamRelationReference> relRef, std::vector<std::unique_ptr<RamExpression>> vals)
44  : relationRef(std::move(relRef)), values(std::move(vals)) {
45  assert(relationRef != nullptr && "Relation reference is a nullptr");
46  for (const auto& v : values) {

48  }

62  return toPtrVector(values);
63  }

66  std::vector<const RamNode*> res = {relationRef.get()};
67  for (const auto& cur : values) {

70  return res;
71  }
