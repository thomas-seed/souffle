234 // Retrieves the <expr1> <= Tuple[level, element] <= <expr2> part of the constraint as a pair { <expr1>,
235 // <expr2> }
236 ExpressionPair MakeIndexTransformer::getLowerUpperExpression(
237  RamCondition* c, size_t& element, int identifier) {
238  if (auto* binRelOp = dynamic_cast<RamConstraint*>(c)) {
239  if (isEqConstraint(binRelOp->getOperator())) {
240  if (const auto* lhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getLHS())) {
241  const RamExpression* rhs = &binRelOp->getRHS();
242  if (lhs->getTupleId() == identifier && rla->getLevel(rhs) < identifier) {
243  element = lhs->getElement();
244  return {clone(rhs), clone(rhs)};
245  }
246  }
247  if (const auto* rhs = dynamic_cast<const RamTupleElement*>(&binRelOp->getRHS())) {
248  const RamExpression* lhs = &binRelOp->getLHS();
249  if (rhs->getTupleId() == identifier && rla->getLevel(lhs) < identifier) {
250  element = rhs->getElement();
251  return {clone(lhs), clone(lhs)};
252  }
253  }
254  } else if (isIneqSigned(binRelOp->getOperator())) {
255  return getSignedExpressionPair(binRelOp, element, identifier);
256  } else if (isIneqUnsigned(binRelOp->getOperator())) {
257  return getUnsignedExpressionPair(binRelOp, element, identifier);
258  } else if (isIneqFloat(binRelOp->getOperator())) {
259  return getFloatExpressionPair(binRelOp, element, identifier);
260  }
261  }
262  return {std::make_unique<RamUndefValue>(), std::make_unique<RamUndefValue>()};
263 }

265 std::unique_ptr<RamCondition> MakeIndexTransformer::constructPattern(
266  const std::vector<std::string>& attributeTypes, RamPattern& queryPattern, bool& indexable,
267  std::vector<std::unique_ptr<RamCondition>> conditionList, int identifier) {
268  // Remaining conditions which cannot be handled by an index
269  std::unique_ptr<RamCondition> condition;
270  auto addCondition = [&](std::unique_ptr<RamCondition> c) {
271  if (condition != nullptr) {
272  condition = std::make_unique<RamConjunction>(std::move(condition), std::move(c));
273  } else {
274  condition = std::move(c);
275  }
276  };
277 
278  // Build query pattern and remaining condition
279  for (auto& cond : conditionList) {
280  size_t element = 0;
281  std::unique_ptr<RamExpression> lowerExpression;
282  std::unique_ptr<RamExpression> upperExpression;
283  std::tie(lowerExpression, upperExpression) = getLowerUpperExpression(cond.get(), element, identifier);
284 
285  // we have new bounds if at least one is defined
286  if (!isRamUndefValue(lowerExpression.get()) || !isRamUndefValue(upperExpression.get())) {
287  // if no previous bounds are set then just assign them, consider both bounds to be set (but not
288  // necessarily defined) in all remaining cases
289  auto type = attributeTypes[element];
290  indexable = true;
291  if (isRamUndefValue(queryPattern.first[element].get()) &&
292  isRamUndefValue(queryPattern.second[element].get())) {
293  queryPattern.first[element] = std::move(lowerExpression);
294  queryPattern.second[element] = std::move(upperExpression);
295  // if lower bound is undefined and we have a new lower bound then assign it
296  } else if (isRamUndefValue(queryPattern.first[element].get()) &&
297  !isRamUndefValue(lowerExpression.get()) && isRamUndefValue(upperExpression.get())) {
298  queryPattern.first[element] = std::move(lowerExpression);
299  // if upper bound is undefined and we have a new upper bound then assign it
300  } else if (isRamUndefValue(queryPattern.second[element].get()) &&
301  isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
302  queryPattern.second[element] = std::move(upperExpression);
303  // if both bounds are defined ...
304  // and equal then we have a previous equality constraint i.e. Tuple[level, element] = <expr1>
305  } else if (!isRamUndefValue(queryPattern.first[element].get()) &&
306  !isRamUndefValue(queryPattern.second[element].get()) &&
307  (*(queryPattern.first[element]) == *(queryPattern.second[element]))) {
308  // new equality constraint i.e. Tuple[level, element] = <expr2>
309  // simply hoist <expr1> = <expr2> to the outer loop
310  if (!isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
311  // FIXME: `FEQ` handling; need to know if the expr is a float exp or not
312  addCondition(std::make_unique<RamConstraint>(getEqConstraint(type),
313  souffle::clone(queryPattern.first[element]), std::move(lowerExpression)));
314  }
315  // new lower bound i.e. Tuple[level, element] >= <expr2>
316  // we need to hoist <expr1> >= <expr2> to the outer loop
317  else if (!isRamUndefValue(lowerExpression.get()) && isRamUndefValue(upperExpression.get())) {
318  addCondition(std::make_unique<RamConstraint>(getGreaterEqualConstraint(type),
319  souffle::clone(queryPattern.first[element]), std::move(lowerExpression)));
320  }
321  // new upper bound i.e. Tuple[level, element] <= <expr2>
322  // we need to hoist <expr1> <= <expr2> to the outer loop
323  else if (isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get())) {
324  addCondition(std::make_unique<RamConstraint>(getLessEqualConstraint(type),
325  souffle::clone(queryPattern.first[element]), std::move(upperExpression)));
326  }
327  // if either bound is defined but they aren't equal we must consider the cases for updating
328  // them note that at this point we know that if we have a lower/upper bound it can't be the
329  // first one
330  } else if (!isRamUndefValue(queryPattern.first[element].get()) ||
331  !isRamUndefValue(queryPattern.second[element].get())) {
332  // if we have a new equality constraint and previous inequality constraints
333  if (!isRamUndefValue(lowerExpression.get()) && !isRamUndefValue(upperExpression.get()) &&
334  *lowerExpression == *upperExpression) {
335  // if Tuple[level, element] >= <expr1> and we see Tuple[level, element] = <expr2>
336  // need to hoist <expr2> >= <expr1> to the outer loop
337  if (!isRamUndefValue(queryPattern.first[element].get())) {
338  addCondition(std::make_unique<RamConstraint>(getGreaterEqualConstraint(type),
339  souffle::clone(lowerExpression), std::move(queryPattern.first[element])));
340  }
341  // if Tuple[level, element] <= <expr1> and we see Tuple[level, element] = <expr2>
342  // need to hoist <expr2> <= <expr1> to the outer loop
343  if (!isRamUndefValue(queryPattern.second[element].get())) {
344  addCondition(std::make_unique<RamConstraint>(getLessEqualConstraint(type),
345  souffle::clone(upperExpression), std::move(queryPattern.second[element])));
346  }
347  // finally replace bounds with equality constraint
348  queryPattern.first[element] = std::move(lowerExpression);
349  queryPattern.second[element] = std::move(upperExpression);
350  // if we have a new lower bound
351  } else if (!isRamUndefValue(lowerExpression.get())) {
352  // we want the tightest lower bound so we take the max
353  std::vector<std::unique_ptr<RamExpression>> maxArguments;
354  maxArguments.push_back(std::move(queryPattern.first[element]));
355  maxArguments.push_back(std::move(lowerExpression));
356 
357  queryPattern.first[element] =
358  std::make_unique<RamIntrinsicOperator>(getMaxOp(type), std::move(maxArguments));
359  // if we have a new upper bound
360  } else if (!isRamUndefValue(upperExpression.get())) {
361  // we want the tightest upper bound so we take the min
362  std::vector<std::unique_ptr<RamExpression>> minArguments;
363  minArguments.push_back(std::move(queryPattern.second[element]));
364  minArguments.push_back(std::move(upperExpression));
365 
366  queryPattern.second[element] =
367  std::make_unique<RamIntrinsicOperator>(getMinOp(type), std::move(minArguments));
368  }
369  }
370  } else {
371  addCondition(std::move(cond));
372  }
373  }
374 
375  // Avoid null-pointers for condition and query pattern
376  if (condition == nullptr) {
377  condition = std::make_unique<RamTrue>();
378  }
379  return condition;
380 }

382 std::unique_ptr<RamOperation> MakeIndexTransformer::rewriteAggregate(const RamAggregate* agg) {
383  if (dynamic_cast<const RamTrue*>(&agg->getCondition()) == nullptr) {
384  const RamRelation& rel = agg->getRelation();
385  int identifier = agg->getTupleId();
386  RamPattern queryPattern;
387  for (unsigned int i = 0; i < rel.getArity(); ++i) {
388  queryPattern.first.push_back(std::make_unique<RamUndefValue>());
389  queryPattern.second.push_back(std::make_unique<RamUndefValue>());
390  }
391 
392  bool indexable = false;
393  std::unique_ptr<RamCondition> condition = constructPattern(rel.getAttributeTypes(), queryPattern,
394  indexable, toConjunctionList(&agg->getCondition()), identifier);
395  if (indexable) {
396  return std::make_unique<RamIndexAggregate>(souffle::clone(&agg->getOperation()),
397  agg->getFunction(), std::make_unique<RamRelationReference>(&rel),
398  souffle::clone(&agg->getExpression()), std::move(condition), std::move(queryPattern),
399  agg->getTupleId());
400  }
401  }
402  return nullptr;

458 bool MakeIndexTransformer::makeIndex(RamProgram& program) {
459  bool changed = false;
460  visitDepthFirst(program, [&](const RamQuery& query) {
461  std::function<std::unique_ptr<RamNode>(std::unique_ptr<RamNode>)> scanRewriter =
462  [&](std::unique_ptr<RamNode> node) -> std::unique_ptr<RamNode> {
463  if (const RamScan* scan = dynamic_cast<RamScan*>(node.get())) {
464  if (scan->getRelation().getRepresentation() != RelationRepresentation::INFO) {
465  if (std::unique_ptr<RamOperation> op = rewriteScan(scan)) {
466  changed = true;
467  node = std::move(op);
468  }
469  }
470  } else if (const RamIndexScan* iscan = dynamic_cast<RamIndexScan*>(node.get())) {
471  if (std::unique_ptr<RamOperation> op = rewriteIndexScan(iscan)) {
472  changed = true;
473  node = std::move(op);
474  }
475  } else if (const RamAggregate* agg = dynamic_cast<RamAggregate*>(node.get())) {
476  if (std::unique_ptr<RamOperation> op = rewriteAggregate(agg)) {
477  changed = true;
478  node = std::move(op);
479  }
480  }
481  node->apply(makeLambdaRamMapper(scanRewriter));
482  return node;
483  };
484  const_cast<RamQuery*>(&query)->apply(makeLambdaRamMapper(scanRewriter));
485  });
486  return changed;
487 }
