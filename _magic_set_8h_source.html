44 /**

47  * (1) NormaliseDatabaseTransformer, for assumptions to hold

245  /** Add an adornment to the ToDo queue if it hasn't been processed before. */
246  void queueAdornment(const AstQualifiedName& relName, const std::string& adornmentMarker) {
247  auto adornmentID = getAdornmentID(relName, adornmentMarker);
248  if (!contains(headAdornmentsSeen, adornmentID)) {
249  headAdornmentsToDo.insert(std::make_pair(relName, adornmentMarker));
250  headAdornmentsSeen.insert(adornmentID);
251  }
252  }
253 
254  /** Check if any more relations need to be adorned. */
255  bool hasAdornmentToProcess() const {
256  return !headAdornmentsToDo.empty();
257  }
258 
259  /** Pop off the next predicate adornment to process. **/
260  adorned_predicate nextAdornmentToProcess() {
261  assert(hasAdornmentToProcess() && "no adornment to pop");
262  auto headAdornment = *(headAdornmentsToDo.begin());
263  headAdornmentsToDo.erase(headAdornmentsToDo.begin());
264  return headAdornment;
265  }
