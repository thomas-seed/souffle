34 /**
35  * @class RamLogRelationTimer
36  * @brief Execution time logger for a statement
37  *
38  * Logs the execution time of a statement. Before and after
39  * the execution of the logging statement the wall-clock time
40  * is taken to compute the time duration for the statement.
41  * Duration and logging message is printed after the execution
42  * of the statement.
43  *
44  * For example:
45  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
46  * START_TIMER ON A "file.dl [8:1-8:8]\;"
47  * ...
48  * END_TIMER
49  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
50  */
51 class RamLogRelationTimer : public RamRelationStatement, public RamAbstractLog {
52 public:
53  RamLogRelationTimer(
54  std::unique_ptr<RamStatement> stmt, std::string msg, std::unique_ptr<RamRelationReference> relRef)
55  : RamRelationStatement(std::move(relRef)), RamAbstractLog(std::move(stmt), std::move(msg)) {}
56 
57  std::vector<const RamNode*> getChildNodes() const override {
58  std::vector<const RamNode*> res = RamRelationStatement::getChildNodes();
59  res.push_back(RamAbstractLog::getChildNodes().at(0));
60  return res;
61  }
62 
63  RamLogRelationTimer* clone() const override {
64  return new RamLogRelationTimer(souffle::clone(statement), message, souffle::clone(relationRef));
65  }
66 
67  void apply(const RamNodeMapper& map) override {
68  RamRelationStatement::apply(map);
69  RamAbstractLog::apply(map);
70  }
71 
72 protected:
73  void print(std::ostream& os, int tabpos) const override {
74  os << times(" ", tabpos) << "START_TIMER ON " << getRelation().getName() << " \""
75  << stringify(message) << "\"" << std::endl;
76  RamStatement::print(statement.get(), os, tabpos + 1);
77  os << times(" ", tabpos) << "END_TIMER" << std::endl;
78  }
79 };
