31 bool ADTtoRecordsTransformer::transform(AstTranslationUnit& tu) {
32  struct ADTsFuneral : public AstNodeMapper {
33  mutable bool changed{false};
34  AstTranslationUnit& tu;
35  const SumTypeBranchesAnalysis& sumTypesBranches = *tu.getAnalysis<SumTypeBranchesAnalysis>();
36 
37  ADTsFuneral(AstTranslationUnit& tu) : tu(tu){};
38 
39  Own<AstNode> operator()(Own<AstNode> node) const override {
40  // Rewrite sub-expressions first
41  node->apply(*this);
42 
43  if (!isA<AstBranchInit>(node)) {
44  return node;
45  }
46 
47  changed = true;
48 
49  auto& adt = *as<AstBranchInit>(node);
50 
51  auto& type = sumTypesBranches.unsafeGetType(adt.getConstructor());
52  assert(isA<AlgebraicDataType>(type));
53 
54  auto& branches = as<AlgebraicDataType>(type)->getBranches();
55 
56  // Find branch ID.
57  AlgebraicDataType::Branch searchDummy{adt.getConstructor(), {}};
58  auto iterToBranch = std::lower_bound(branches.begin(), branches.end(), searchDummy,
59  [](const AlgebraicDataType::Branch& left, const AlgebraicDataType::Branch& right) {
60  return left.name < right.name;
61  });
62 
63  // Branch id corresponds to the position in lexicographical ordering.
64  auto branchID = std::distance(std::begin(branches), iterToBranch);
65 
66  // Collect branch arguments
67  VecOwn<AstArgument> branchArguments;
68  for (auto* arg : adt.getArguments()) {
69  branchArguments.emplace_back(arg->clone());
70  }
71 
72  // Store branch arguments as record [branch_args...]
73  auto branchArgsAsRecord = mk<AstArgument, AstRecordInit>(std::move(branchArguments));
74 
75  // Arguments for the resulting record [branch_id, [branch_args...]].
76  VecOwn<AstArgument> finalRecordArgs;
77 
78  finalRecordArgs.push_back(mk<AstArgument, AstNumericConstant>(branchID));
79  finalRecordArgs.push_back(std::move(branchArgsAsRecord));
80 
81  return mk<AstRecordInit>(std::move(finalRecordArgs), adt.getSrcLoc());
82  }
83  };
84 
85  ADTsFuneral mapper(tu);
86  tu.getProgram()->apply(mapper);
87  return mapper.changed;
88 }
