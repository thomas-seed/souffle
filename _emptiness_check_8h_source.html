48  assert(relationRef != nullptr && "Relation reference is a nullptr");
49  }

51  /** @brief Get relation */

53  return *relationRef->get();
54  }

56  std::vector<const RamNode*> getChildNodes() const override {

61  return new RamEmptinessCheck(souffle::clone(relationRef));
62  }
63

66  }
67 
68 protected:
