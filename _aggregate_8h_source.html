38 /**
39  * @class RamAggregate
40  * @brief Aggregation function applied on some relation
41  *
42  * For example:
43  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
44  * t0.0 = COUNT FOR ALL t0 IN A
45  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~
46  * Applies the function COUNT to determine the number
47  * of elements in A.
48  */
49 class RamAggregate : public RamRelationOperation, public RamAbstractAggregate {
50 public:
51  RamAggregate(std::unique_ptr<RamOperation> nested, AggregateOp fun,
52  std::unique_ptr<RamRelationReference> relRef, std::unique_ptr<RamExpression> expression,
53  std::unique_ptr<RamCondition> condition, int ident)
54  : RamRelationOperation(std::move(relRef), ident, std::move(nested)),
55  RamAbstractAggregate(fun, std::move(expression), std::move(condition)) {}

57  std::vector<const RamNode*> getChildNodes() const override {
58  auto res = RamRelationOperation::getChildNodes();
59  auto children = RamAbstractAggregate::getChildNodes();
60  res.insert(res.end(), children.begin(), children.end());
61  return res;
62  }

67  }

70  RamRelationOperation::apply(map);

80  os << "FOR ALL t" << getTupleId() << " ∈ " << getRelation().getName();
