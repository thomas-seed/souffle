82  class Hasher {
83  public:
84  size_t operator()(const SearchSignature& searchSignature) const {
85  std::size_t seed = searchSignature.arity();
86  for (auto& constraint : searchSignature.constraints) {
87  seed ^= static_cast<size_t>(constraint) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
88  }
89  return seed;
90  }

102  *
103  * This class is a helper class for RamIndex.
104  *
105  * This implements a standard maximum matching algorithm for a bi-partite graph
106  * also known as a marriage problem. Given a set of edges in a bi-partite graph
107  * select a subset of edges that each node in the bi-partite graph has at most
108  * one adjacent edge associated with.
109  *
110  * The nodes of the bi-partite graph represent index-signatures stemming from
111  * RAM operations and RAM existence checks for a relation. A relation between
112  * two nodes represent whether an index operation subsumes another operation.
113  *

115  */

125  * graph maps to another node. If no map exist for a node, there is no adjacent edge

127  */

316  /** @Brief count the number of constraints in key */
317  static size_t card(SearchSignature cols) {
318  size_t sz = 0;
319  for (size_t i = 0; i < cols.arity(); i++) {

321  sz++;
322  }
323  }

327  /** @Brief maps search columns to an lexicographical order (labeled by a number) */
328  int map(SearchSignature cols) const {
329  assert(orders.size() == chainToOrder.size() && "Order and Chain Sizes do not match!!");
330  int i = 0;
331  for (auto it = chainToOrder.begin(); it != chainToOrder.end(); ++it, ++i) {
332  if (std::find(it->begin(), it->end(), cols) != it->end()) {
333  assert((size_t)i < orders.size());
334  return i;
335  }
336  }
337  fatal("cannot find matching lexicographical order");
338  }

340  /** @Brief insert an index based on the delta */
341  void insertIndex(LexOrder& ids, SearchSignature delta) {
342  LexOrder backlog; // add inequalities at the end
343  for (size_t pos = 0; pos < delta.arity(); pos++) {
344  if (delta[pos] == AttributeConstraint::Equal) {
345  ids.push_back(pos);
346  } else if (delta[pos] == AttributeConstraint::Inequal) {
347  backlog.push_back(pos);
348  }
349  }
350  ids.insert(ids.end(), backlog.begin(), backlog.end());
351  }
