39 /**
40  * @class RamParallelIndexAggregate
41  * @brief Aggregate over values of a relation using an index in parallel
42  *
43  * For example:
44  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
45  * t0.0=sum t0.1 SEARCH t0 ∈ S ON INDEX t0.0 = number(1)
46  * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
47  */
48 class RamParallelIndexAggregate : public RamIndexAggregate, public RamAbstractParallel {
49 public:
50  RamParallelIndexAggregate(std::unique_ptr<RamOperation> nested, AggregateOp fun,
51  std::unique_ptr<RamRelationReference> relRef, std::unique_ptr<RamExpression> expression,
52  std::unique_ptr<RamCondition> condition, RamPattern queryPattern, int ident)
53  : RamIndexAggregate(std::move(nested), fun, std::move(relRef), std::move(expression),
54  std::move(condition), std::move(queryPattern), ident) {}
55 
56  RamParallelIndexAggregate* clone() const override {
57  RamPattern pattern;
58  for (const auto& i : queryPattern.first) {
59  pattern.first.emplace_back(i->clone());
60  }
61  for (const auto& i : queryPattern.second) {
62  pattern.second.emplace_back(i->clone());
63  }
64  return new RamParallelIndexAggregate(souffle::clone(&getOperation()), function,
65  souffle::clone(relationRef), souffle::clone(expression), souffle::clone(condition),
66  std::move(pattern), getTupleId());
67  }
68 
69 protected:
70  void print(std::ostream& os, int tabpos) const override {
71  os << times(" ", tabpos);
72  os << "PARALLEL t" << getTupleId() << ".0=";
73  RamAbstractAggregate::print(os, tabpos);
74  os << "SEARCH t" << getTupleId() << " ∈ " << getRelation().getName();
75  printIndex(os);
76  if (!isRamTrue(condition.get())) {
77  os << " WHERE " << getCondition();
78  }
79  os << std::endl;
80  RamIndexOperation::print(os, tabpos + 1);
81  }
82 };
